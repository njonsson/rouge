collect

Object collect(<Function> block)
    Creates a different array of the same size. Invokes block once for each element in the array, passing that element as an argument. 
     var array = ['foo', 'bar', 'baz'];
 var result = array.collect(function(item) {
   return item + '!';
 });
 result // => ['foo!', 'bar!', 'baz!']
 array  // => ['foo', 'bar', 'baz']
    Parameters:
      block - The function to execute. Should have one parameter
    Returns:
      An array containing the values returned by block

detect

Object detect(<Function> ifNone, <Function> block)
    Finds an element in the array. Invokes block once for each element in the array, passing that element as an argument. 
     var array = ['foo', 'bar', 'baz'];
 var result = array.collect(function(item) {
   return item == 'bar';
 });
 result // => 'bar'
 array  // => ['foo', 'bar', 'baz']
    With the optional ifNone argument: 
     var array = ['foo', 'bar', 'baz'];
 var result = array.collect(function() { return 'nothing here'; }, function(item) {
   return item == 'bizzle';
 });
 result // => 'nothing here'
 array  // => ['foo', 'bar', 'baz']
    Parameters:
      ifNone - (optional) A function that returns the value to be returned by #detect if there are no matches
      block - The function to execute. Should have one parameter and return either true or false
    Returns:
      The first element for which block returns true. If block never returns true and ifNone is specified, then the return value of a call to ifNone is used. If block never returns true and ifNone is not specified, then null is used

each

Object each(<Function> block)
    Invokes block once for each element in the array, passing that element as an argument. 
     var array = ['foo', 'bar', 'baz'];
 array.each(function(item) {
   alert(item);
 });
    opens three alert boxes:
    • foo
    • bar
    • baz
    Parameters:
      block - The function to execute. Should have one parameter
    Returns:
      The array

eachWithIndex

Object eachWithIndex(<Function> block)
    Invokes block once for each element in the array, passing that element and its index as arguments. 
     var array = ['foo', 'bar', 'baz'];
 array.eachWithIndex(function(item, index) {
   alert(index + ': ' + item);
 });
    opens three alert boxes:
    • 0: foo
    • 1: bar
    • 2: baz
    Parameters:
      block - The function to execute. Should have two parameters
    Returns:
      The array
